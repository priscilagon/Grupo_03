CREATE DATABASE IF NOT EXISTS vacinacao_db;
USE vacinacao_db;

-- ==========================================================
-- 1. CRIAÇÃO DAS TABELAS (DDL)
-- ==========================================================

CREATE TABLE usuario(
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    nome_usuario VARCHAR(100) NOT NULL,
    login_usuario VARCHAR(100) NOT NULL UNIQUE,
    senha_usuario VARCHAR(255) NOT NULL,
    perfil_usuario ENUM('Admin', 'Enfermeiro', 'Recepcionista', 'Logística') NOT NULL,
    status_perfil ENUM('ativo', 'inativo') DEFAULT 'ativo'
);

CREATE TABLE faixa_etaria(
    id_faixa INT PRIMARY KEY AUTO_INCREMENT,
    descricao VARCHAR(50) NOT NULL,
    idade_min INT NOT NULL,
    idade_max INT NOT NULL
);

CREATE TABLE cargo_profissional(
    id_cargo INT PRIMARY KEY AUTO_INCREMENT,
    nome_cargo VARCHAR(60) NOT NULL UNIQUE
);

CREATE TABLE posto_saude(
    id_posto INT PRIMARY KEY AUTO_INCREMENT,
    nome_posto VARCHAR(100) NOT NULL,
    logradouro VARCHAR(100) NOT NULL,
    numero VARCHAR(10) NOT NULL,
    bairro VARCHAR(50) NOT NULL, 
    cidade VARCHAR(50) DEFAULT 'Manaus',
    cep VARCHAR(9) NOT NULL,
    telefone VARCHAR(20) NOT NULL
);

CREATE TABLE paciente(
    id_paciente INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(120) NOT NULL,
    cpf VARCHAR(14) UNIQUE NOT NULL,
    cns VARCHAR(15) UNIQUE NOT NULL,
    data_nascimento DATE NOT NULL,
    sexo ENUM('M', 'F') NOT NULL,
    telefone VARCHAR(20) NOT NULL,
    logradouro VARCHAR(100) NOT NULL,
    numero VARCHAR(10) NOT NULL,
    bairro VARCHAR(50) NOT NULL, 
    cidade VARCHAR(50) DEFAULT 'Manaus',
    cep VARCHAR(9) NOT NULL,
    id_faixa INT NOT NULL,
    FOREIGN KEY(id_faixa) REFERENCES faixa_etaria(id_faixa)
);

CREATE TABLE profissional(
    id_profissional INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(120) NOT NULL,
    cpf VARCHAR(14) NOT NULL UNIQUE,
    registro_conselho VARCHAR(30) NULL,
    id_cargo INT NOT NULL,
    id_usuario INT UNIQUE NOT NULL, 
    id_posto INT NOT NULL,
    FOREIGN KEY(id_cargo) REFERENCES cargo_profissional(id_cargo),
    FOREIGN KEY(id_usuario) REFERENCES usuario(id_usuario),
    FOREIGN KEY(id_posto) REFERENCES posto_saude(id_posto)
);

CREATE TABLE vacina(
    id_vacina INT PRIMARY KEY AUTO_INCREMENT,
    nome_vacina VARCHAR(100) NOT NULL,
    fabricante_vacina VARCHAR(100) NOT NULL,
    doses_necessarias INT DEFAULT 1,
    intervalo_doses_dias INT NOT NULL, 
    descricao_vacina TEXT NOT NULL
);

CREATE TABLE lote_vacina(
    id_lote INT PRIMARY KEY AUTO_INCREMENT,
    numero_lote VARCHAR(50) NOT NULL UNIQUE,
    data_validade DATE NOT NULL,
    quantidade_inicial INT NOT NULL,
    quantidade_atual INT NOT NULL,
    id_vacina INT NOT NULL,
    id_posto INT NOT NULL,
    FOREIGN KEY(id_vacina) REFERENCES vacina(id_vacina),
    FOREIGN KEY(id_posto) REFERENCES posto_saude(id_posto)
);

CREATE TABLE aplicacao_vacina(
    id_aplicacao INT PRIMARY KEY AUTO_INCREMENT,
    data_aplicacao DATETIME DEFAULT CURRENT_TIMESTAMP,
    dose_aplicada VARCHAR(20) NOT NULL, 
    id_paciente INT NOT NULL,
    id_profissional INT NOT NULL,
    id_lote INT NOT NULL,
    id_posto INT NOT NULL,
    FOREIGN KEY(id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY(id_profissional) REFERENCES profissional(id_profissional),
    FOREIGN KEY(id_lote) REFERENCES lote_vacina(id_lote),
    FOREIGN KEY(id_posto) REFERENCES posto_saude(id_posto)
);

CREATE TABLE agendamento(
    id_agendamento INT PRIMARY KEY AUTO_INCREMENT,
    data_agendada DATE NOT NULL,
    status_agendamento ENUM('pendente', 'concluido', 'faltoso') DEFAULT 'pendente',
    id_paciente INT NOT NULL,
    id_vacina INT NOT NULL,
    id_posto INT NOT NULL,
    FOREIGN KEY(id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY(id_vacina) REFERENCES vacina(id_vacina),
    FOREIGN KEY(id_posto) REFERENCES posto_saude(id_posto)
);

CREATE TABLE reacao_adversa(
    id_reacao INT PRIMARY KEY AUTO_INCREMENT,
    descricao_reacao TEXT NOT NULL,
    gravidade ENUM('leve', 'moderada', 'grave') DEFAULT 'leve',
    data_relato DATE NOT NULL,
    id_aplicacao INT NOT NULL,
    FOREIGN KEY(id_aplicacao) REFERENCES aplicacao_vacina(id_aplicacao)
);

-- =========================================================================================
-- TRIGGER 1: Atualizar estoque após aplicação
-- =========================================================================================
-- OBJETIVO: Garantir que o inventário de vacinas seja atualizado em tempo real.
-- LOGICA: Toda vez que um registro de aplicação de vacina é CRIADO (AFTER INSERT),
--         a trigger localiza o lote utilizado e subtrai 1 da 'quantidade_atual'.
-- POR QUE USAR: Evita que o estoque fique "mentiroso" e permite o controle de falta de insumos.
-- =========================================================================================
CREATE TRIGGER tr_atualizar_estoque_apos_aplicacao
AFTER INSERT ON aplicacao_vacina
FOR EACH ROW
BEGIN
    UPDATE lote_vacina
    SET quantidade_atual = quantidade_atual - 1
    WHERE id_lote = NEW.id_lote;
END;
//

-- =========================================================================================
-- TRIGGER 2: Impedir uso de lote vencido
-- =========================================================================================
-- OBJETIVO: Segurança do paciente e conformidade sanitária.
-- LOGICA: ANTES (BEFORE) de salvar a aplicação, o sistema consulta a validade do lote.
--         Se a data de validade for menor que a data de hoje (CURDATE), o MySQL gera um
--         erro (SIGNAL SQLSTATE) e cancela a inserção dos dados.
-- POR QUE USAR: Impede erros humanos críticos, como vacinar alguém com produto vencido.
-- =========================================================================================
CREATE TRIGGER tr_impedir_lote_vencido
BEFORE INSERT ON aplicacao_vacina
FOR EACH ROW
BEGIN
    DECLARE v_validade DATE;

    SELECT data_validade INTO v_validade 
    FROM lote_vacina 
    WHERE id_lote = NEW.id_lote;

    IF v_validade < CURDATE() THEN
        SIGNAL SQLSTATE '45000' 
        SET MESSAGE_TEXT = 'Operação Cancelada: O lote selecionado está vencido.';
    END IF;
END;
//

-- =========================================================================================
-- TRIGGER 3 (BÔNUS): Validação do Conselho Profissional
-- =========================================================================================
-- OBJETIVO: Integridade de dados para cargos administrativos vs. assistenciais.
-- LOGICA: Ao cadastrar um profissional, a trigger verifica o nome do seu cargo.
--         Se o cargo NÃO for de uma categoria que exija registro de classe (ex: Motorista),
--         a trigger força o campo 'registro_conselho' para NULL, mesmo que o usuário 
--         tenha tentado preencher algo.
-- POR QUE USAR: Mantém o banco limpo, evitando registros de "conselho" fantasmas para
--              cargos que não possuem conselho de classe.
-- =========================================================================================
CREATE TRIGGER tr_validar_conselho_insert
BEFORE INSERT ON profissional
FOR EACH ROW
BEGIN
    DECLARE v_nome_cargo VARCHAR(60);
    
    SELECT nome_cargo INTO v_nome_cargo 
    FROM cargo_profissional 
    WHERE id_cargo = NEW.id_cargo;

    IF v_nome_cargo NOT IN (
        'Assistente Social', 'Auxiliar de Saúde Bucal', 'Bioquímico', 
        'Enfermeiro(a)', 'Farmacêutico(a)', 'Fisioterapeuta', 
        'Fonoaudiólogo(a)', 'Infectologista', 'Médico(a) Clínico', 
        'Médico(a) Pediatra', 'Nutricionista', 'Odontólogo(a)', 'Psicólogo(a)'
    ) THEN
        SET NEW.registro_conselho = NULL;
    END IF;
END;

-- =========================================================================================
-- PROCEDURE 1: Previsão da Próxima Dose
-- =========================================================================================
-- OBJETIVO: Calcular automaticamente quando o paciente deve retornar.
-- PARÂMETROS: Recebe o ID do paciente e o ID da vacina.
-- FUNCIONAMENTO:
--   1. Consulta na tabela 'vacina' o intervalo de dias e o total de doses necessárias.
--   2. Conta quantas vezes aquele paciente já tomou aquela vacina específica.
--   3. Se houver doses pendentes, soma o intervalo à data da última aplicação (DATE_ADD).
--   4. Retorna mensagens informativas se o esquema estiver completo ou não iniciado.
-- POR QUE USAR: Automatiza a agenda da equipe de enfermagem e evita erros de cálculo manual.
-- =========================================================================================
CREATE PROCEDURE sp_prever_proxima_dose(
    IN p_id_paciente INT,
    IN p_id_vacina INT
)
BEGIN
    DECLARE v_intervalo INT;
    DECLARE v_doses_totais INT;
    DECLARE v_doses_aplicadas INT;
    DECLARE v_ultima_aplicacao DATE;
    DECLARE v_proxima_data DATE;

    -- Pega as regras da vacina
    SELECT intervalo_doses_dias, doses_necessarias 
    INTO v_intervalo, v_doses_totais
    FROM vacina 
    WHERE id_vacina = p_id_vacina;

    -- Verifica o histórico do paciente para esta vacina
    SELECT COUNT(*), MAX(data_aplicacao)
    INTO v_doses_aplicadas, v_ultima_aplicacao
    FROM aplicacao_vacina av
    JOIN lote_vacina lv ON av.id_lote = lv.id_lote
    WHERE av.id_paciente = p_id_paciente AND lv.id_vacina = p_id_vacina;

    -- Validações de retorno
    IF v_doses_totais = 1 AND v_doses_aplicadas >= 1 THEN
        SELECT 'Esquema de dose única já concluído.' AS Mensagem;
        
    ELSEIF v_doses_aplicadas = 0 THEN
        SELECT 'Paciente ainda não recebeu a primeira dose.' AS Mensagem;
        
    ELSEIF v_doses_aplicadas >= v_doses_totais THEN
        SELECT 'Esquema vacinal completo.' AS Mensagem;
        
    ELSE
        -- Cálculo da data sugerida
        SET v_proxima_data = DATE_ADD(v_ultima_aplicacao, INTERVAL v_intervalo DAY);
        SELECT 
            v_proxima_data AS data_sugerida,
            (v_doses_aplicadas + 1) AS dose_sequencial,
            v_intervalo AS dias_de_intervalo;
    END IF;
END;
