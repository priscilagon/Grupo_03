-- Consultas DQL (Data Query Language)
-- consulta 01
SELECT ps.nome_posto, COUNT(av.id_aplicacao) AS total_vacinas
FROM posto_saude ps
JOIN aplicacao_vacina av ON ps.id_posto = av.id_posto
GROUP BY ps.nome_posto
HAVING total_vacinas > 2; 

-- consulta 02
SELECT nome, data_nascimento, 
       TIMESTAMPDIFF(YEAR, data_nascimento, CURDATE()) AS idade_atual
FROM paciente
ORDER BY idade_atual DESC;
